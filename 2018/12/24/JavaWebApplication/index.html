JavaWeb应用–实现前端后台简单交互

1. 实现功能
  本文实现的功能：登陆界面输入用户名和密码。如果用户名或者密码输入正确，则跳转到正确页面；否则，跳转到失败页面。

**注意** 用户名：admin  密码：123456（为了方便，没有建立用户数据库库查询，而是直接定义好用户名与密码) **2.实现步骤**   需要准备的环境：Eclipse **2.1 创建Dynamic Web Project**   工程路径：**D:\WorkSpace\Ecplise-WorkSpace\Classes\Login**   创建方式：Eclipse->File->new->Dynamic Web Project->输入工程名   工程目录说明如下图所示:
**2.2 前端页面（登陆、登陆成功、登陆失败）**   路径: **D:\WorkSpace\Ecplise-WorkSpace\Classes\Login\WebContent\jsp**   2.2.1 登陆页面(login.jsp)   2.2.2 登陆成功页面(loginSuccess.jsp)   2.2.3 登陆失败页面(loginFail.jsp) **2.3 后台服务**   路径: **D:\WorkSpace\Ecplise-WorkSpace\Classes\Login\src**   2.3.1 loginServlet类-获取前端页面用户名和密码，调用loginDeal方法，并跳转至页面（成功或失败页面）   2.3.2 loginDeal类-判断用户名或密码是否正确 **说明：**经过编译的工程，loginServlet.class和loginDeal.class一般默认会放在 .\Login\build\classes路径下，但我们需要将其放在 .\Login\WebContent\WEB-INF\classes路径下,因此需要修改buildpath的默认路径。 **2.4 配置web.xml**   web.xml文件实现对Servlet的URL映射。Web应用的默认URL入口是Web应用的根目录名，eg:对于Login应用，即:/Login **3.代码附录(点击代码名称显示/隐藏）**
  登陆界面代码

login.jsp 

import java.io.IOException;
import javax.servlet.ServletConfig;
import javax.servlet.ServletException;
import javax.servlet.ServletRequest;
import javax.servlet.ServletResponse;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;    
/**
 * @author zhaoyuan
 * @date 2018年12月20日
 * @function 用于获取前端页面（login.jsp）用户输入的用户名和密码，然后调用loginDeal方法进行处理跳转，
 *           若用户名和密码正确则跳转至login_success.jsp页面,否则跳转至login_fail.jsp页面
 */
public class loginServlet implements javax.servlet.Servlet{
    public void doPost(HttpServletRequest request, HttpServletResponse response)throws Exception {
        String username = request.getParameter("username");// 取得用户名
        String password = request.getParameter("password");// 取得密码
        System.out.println("取得用户名和密码");

        loginDeal db = new loginDeal(); // 构建登陆对象
        boolean canLogin = db.LoginDeal(username, password);// 取得用户名和密码
        if (canLogin) {// 根据登陆情况，跳转页面
            System.out.println("用户名和密码正确");
            response.sendRedirect("jsp/login_success.jsp");//地址重定向函数
        } else {
            System.out.println("用户名和密码错误");
            response.sendRedirect("jsp/login_fail.jsp");    
        }    
    }
    public void destroy() {
    }
    public ServletConfig getServletConfig() {
        return null;
    }
    public String getServletInfo() {
        return null;
    }
    public void init(ServletConfig arg0) throws ServletException {
    }
    public void service(ServletRequest request, ServletResponse response) throws ServletException, IOException {
        HttpServletRequest rq = (HttpServletRequest) request;
        HttpServletResponse rs = (HttpServletResponse) response;
        try {
            doPost(rq, rs);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }    
}

  web.xml配置文件

web.xml

<?xml version="1.0" encoding="UTF-8"?>
<web-app xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns="http://xmlns.jcp.org/xml/ns/javaee" xsi:schemaLocation="http://xmlns.jcp.org/xml/ns/javaee http://xmlns.jcp.org/xml/ns/javaee/web-app_3_1.xsd" id="WebApp_ID" version="3.1">
  <display-name>Login</display-name>
  <welcome-file-list>
    <welcome-file>login.html</welcome-file>
    <welcome-file>login.htm</welcome-file>
    <welcome-file>login.jsp</welcome-file>
    <welcome-file>default.html</welcome-file>
    <welcome-file>default.htm</welcome-file>
    <welcome-file>default.jsp</welcome-file>
  </welcome-file-list>
  <servlet>
    <servlet-name>loginServlet</servlet-name>
    <servlet-class>loginServlet</servlet-class>
  </servlet>
  <servlet-mapping>
    <servlet-name>loginServlet</servlet-name>
    <url-pattern>/login</url-pattern>
  </servlet-mapping>
</web-app>
